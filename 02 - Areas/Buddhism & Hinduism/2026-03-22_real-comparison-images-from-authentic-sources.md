downloaded = []
failed = []

img_num = 0
for item in IMAGES:
    if item[0] == "SEC":
        downloaded.append(item)
        continue

    _, label, url = item
    img_num += 1
    fname = f"downloaded_images/{img_num:03d}_{label[:40].replace(' ','_').replace('/','_').replace(':','')}.jpg"

    print(f"  [{img_num:02d}] Downloading: {label[:55]}...")
    try:
        r = requests.get(url, headers=HEADERS, timeout=20)
        if r.status_code == 200 and len(r.content) > 5000:
            img = Image.open(BytesIO(r.content)).convert("RGB")
            img.save(fname, "JPEG", quality=90)
            downloaded.append(("IMG", label, fname))
            print(f"        ✅ Saved ({img.size[0]}x{img.size[1]}px)")
        else:
            print(f"        ❌ Failed (status={r.status_code}, size={len(r.content)})")
            failed.append(label)
    except Exception as e:
        print(f"        ❌ Error: {str(e)[:60]}")
        failed.append(label)
    time.sleep(0.3)

print(f"\n✅ Downloaded: {img_num - len(failed)} images")
print(f"❌ Failed: {len(failed)} images")
if failed:
